<objective>
Add install-time profile selection to the GSD Copilot CLI installer. Since Copilot CLI bakes model IDs into `.agent.md` files at install time (unlike Claude Code's runtime resolution), users must choose a profile when installing. This plan adds:
1. `--profile <quality|balanced|budget>` CLI flag
2. `promptProfile()` interactive function (readline-based, fires after `promptLocation` when Copilot is selected)
3. Wiring of the selected profile down to `installAllRuntimes` ŌåÆ `install()` (which was already updated in Plan 02-01)
4. Write the selected profile to `.planning/config.json` during Copilot install

Purpose: Without profile selection at install time, users have no way to choose which LLM tier to use for GSD subagents in Copilot CLI.

Output: Updated `bin/install.js` with `--profile` flag, `promptProfile()` function, and `.planning/config.json` write during Copilot install.
</objective>

    ```js
    /**
     * Prompt for model profile selection (Copilot installs only).
     * If Copilot is not in the selected runtimes, skips prompt entirely.
     */
    function promptProfile(runtimes, isGlobal, cliProfile) {
      // If profile already set via --profile flag, skip prompt
      if (cliProfile) {
        installAllRuntimes(runtimes, isGlobal, true, cliProfile);
        return;
      }
    
      // If Copilot not selected, profile is irrelevant ŌĆö skip prompt
      if (!runtimes.includes('copilot')) {
        installAllRuntimes(runtimes, isGlobal, true, 'balanced');
        return;
      }
    
      if (!process.stdin.isTTY) {
        console.log(`  ${yellow}Non-interactive terminal detected, defaulting to balanced profile${reset}\n`);
        installAllRuntimes(runtimes, isGlobal, false, 'balanced');
        return;
      }
    
      const rl = readline.createInterface({
        input: process.stdin,
        output: process.stdout
      });
    
      let answered = false;
    
      rl.on('close', () => {
        if (!answered) {
          answered = true;
          console.log(`\n  ${yellow}Installation cancelled${reset}\n`);
          process.exit(0);
        }
      });
    
      console.log(`  ${yellow}Which model profile for Copilot agents?${reset}
      ${dim}(Model IDs are baked into .agent.md files at install time.)${reset}\n
      ${cyan}1${reset}) balanced  ${dim}(default: opus for planners, sonnet for most, haiku for mappers)${reset}
      ${cyan}2${reset}) quality   ${dim}(opus for heavy reasoning agents, sonnet for the rest)${reset}
      ${cyan}3${reset}) budget    ${dim}(sonnet for core agents, haiku for lighter tasks)${reset}
    `);
    
      rl.question(`  Choice ${dim}[1]${reset}: `, (answer) => {
        answered = true;
        rl.close();
        const choice = answer.trim() || '1';
        const profileMap = { '1': 'balanced', '2': 'quality', '3': 'budget' };
        const resolvedProfile = profileMap[choice] || 'balanced';
        installAllRuntimes(runtimes, isGlobal, true, resolvedProfile);
      });
    }
    ```

    Add `writePlanningConfig` helper (near the other utility helpers, before `install()`):

    ```js
    /**
     * Write model_profile to .planning/config.json in the current working directory.
     * Creates the file if absent, updates model_profile if present.
     * Only called during Copilot install (model is baked into .agent.md at install time).
     */
    function writePlanningConfig(profile) {
      const planningDir = path.join(process.cwd(), '.planning');
      const configPath = path.join(planningDir, 'config.json');
      let config = {};
      if (fs.existsSync(configPath)) {
        try {
          config = JSON.parse(fs.readFileSync(configPath, 'utf8'));
        } catch {
          // Malformed JSON ŌĆö start fresh
          config = {};
        }
      }
      config.model_profile = profile;
      // Don't create .planning/ if it doesn't exist ŌĆö only write if dir is present
      // (avoids polluting non-GSD project directories)
      if (fs.existsSync(planningDir)) {
        fs.writeFileSync(configPath, JSON.stringify(config, null, 2) + '\n');
        console.log(`  ${green}Ō£ō${reset} Set model_profile: ${cyan}${profile}${reset} in .planning/config.json`);
      }
    }
    ```

    **Important:** Only write when `.planning/` directory already exists (guard with `fs.existsSync(planningDir)`).
